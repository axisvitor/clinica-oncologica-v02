"""Analyze duplicate migrations in Alembic versions directory"""
import re
from pathlib import Path
from collections import defaultdict

versions_dir = Path("alembic/versions")
migrations = defaultdict(list)

# Group migrations by purpose
for migration_file in versions_dir.glob("*.py"):
    if migration_file.name == "__init__.py":
        continue

    purpose = re.sub(r'^(\d{3}_|20\d{8}_\d{6}_|[a-f0-9]{12}_)', '', migration_file.stem)
    purpose = re.sub(r'_index(es)?$', '', purpose)

    migrations[purpose].append(migration_file.name)

# Report duplicates
print("Migration Duplicate Analysis")
print("=" * 80)

duplicates_found = False
for purpose, files in sorted(migrations.items()):
    if len(files) > 1:
        duplicates_found = True
        print(f"\n❌ DUPLICATE: {purpose}")
        for f in files:
            print(f"   - {f}")

if not duplicates_found:
    print("✅ No duplicates found")

# Report naming convention violations
print("\n\nNaming Convention Analysis")
print("=" * 80)

patterns = {
    "Timestamp": r'^\d{8}_\d{6}_',
    "Sequential": r'^\d{3}_',
    "Autogenerated": r'^[a-f0-9]{12}_',
    "Descriptive": r'^[a-z_]+\.py$',
}

violations = defaultdict(list)
for migration_file in versions_dir.glob("*.py"):
    if migration_file.name == "__init__.py":
        continue

    matched = False
    for pattern_name, pattern in patterns.items():
        if re.match(pattern, migration_file.name):
            violations[pattern_name].append(migration_file.name)
            matched = True
            break

    if not matched:

for pattern, files in sorted(violations.items()):
    print(f"\n{pattern} ({len(files)} files):")
    for f in sorted(files)[:5]:  # Show first 5

    if len(files) > 5:

### Step 2.3: Create Migration Linter

**File**: `backend-hormonia/scripts/lint_migrations.py`

```python
#!/usr/bin/env python3
"""
Migration naming convention linter for Alembic.
Enforces YYYYMMDD_HHMMSS_description.py naming pattern.
"""
import re
import sys
from pathlib import Path
from typing import List, Tuple


# Required pattern for new migrations
REQUIRED_PATTERN = r'^\d{8}_\d{6}_[a-z0-9_]+\.py$'

# Allowed legacy patterns (for backwards compatibility)
LEGACY_PATTERNS = [
    r'^\d{3}_',                    # Old sequential (001_, 002_, etc.)
    r'^[a-f0-9]{12}_',            # Autogenerated hash
    r'^(add|create)_[a-z_]+\.py$', # Descriptive (deprecated)
]


def check_migration_name(filepath: Path) -> Tuple[bool, str]:
    """
    Check if migration filename follows naming convention.

    Args:
        filepath: Path to migration file

    Returns:
        Tuple of (is_valid, message)
    """
    filename = filepath.name

    # Skip special files
    if filename in ('__init__.py', '__pycache__'):
        return True, "Skipped: Special file"

    # Allow merge head migrations
    if 'merge' in filename.lower():
        return True, "Allowed: Merge migration"

    # Check if matches required pattern
    if re.match(REQUIRED_PATTERN, filename):
        return True, "✅ Correct naming"

    # Check if matches legacy pattern
    for pattern in LEGACY_PATTERNS:
        if re.match(pattern, filename):
            return True, "⚠️  Legacy naming (acceptable for old migrations)"

    # Pattern violation
    return False, f"❌ Does not match YYYYMMDD_HHMMSS_description.py pattern"


def lint_all_migrations(versions_dir: Path) -> int:
    """
    Lint all migrations in versions directory.

    Args:
        versions_dir: Path to alembic versions directory

    Returns:
        Number of violations found
    """
    violations = 0
    migration_files = sorted(versions_dir.glob("*.py"))

    print(f"Linting {len(migration_files)} migration files...\n")

    for migration_file in migration_files:
        is_valid, message = check_migration_name(migration_file)

        if not is_valid:
            violations += 1
            print(f"{message}: {migration_file.name}")
        elif message.startswith("⚠️"):
            print(f"{message}: {migration_file.name}")

    return violations


def main():
    """Main entry point for migration linter."""
    versions_dir = Path("alembic/versions")

    if not versions_dir.exists():
        print("❌ Error: alembic/versions directory not found")
        sys.exit(1)

    violations = lint_all_migrations(versions_dir)

    print("\n" + "=" * 80)
    if violations == 0:
        print("✅ All migrations follow naming conventions")
        sys.exit(0)
    else:
        print(f"❌ Found {violations} naming convention violation(s)")
        print("\nNew migrations MUST follow pattern: YYYYMMDD_HHMMSS_description.py")
        print("Example: 20251010_143000_add_patient_consent_tracking.py")
        sys.exit(1)


if __name__ == "__main__":
    main()
```

# Alembic Migration Standards

## Naming Convention (MANDATORY)

All new migrations MUST follow this pattern:
```
YYYYMMDD_HHMMSS_descriptive_action.py
```

### Examples
✅ `20251010_143000_add_patient_consent_tracking.py`
✅ `20251011_090000_create_medication_reminders_table.py`
✅ `20251012_160000_add_timezone_support_to_appointments.py`

❌ `add_patient_consent.py` (no timestamp)
❌ `040_medication_table.py` (old sequential style)
❌ `a3f89c2d1b4e_create_table.py` (autogenerated hash)

## Migration Template

```python
"""Brief one-line description

Revision ID: YYYYMMDD_HHMMSS
Revises: previous_revision
Create Date: YYYY-MM-DD HH:MM:SS

## Purpose
Explain why this migration is needed.

## Changes
- List all tables/columns/indexes being modified
- Include expected impact on queries
- Document any data transformations

## Rollback Considerations
- Explain what happens on downgrade
- Note any data loss risks
- Document manual steps if needed

## Performance Impact
- Estimated execution time: X seconds
- Index build time: Y seconds
- Locks required: [list tables]

Related Issues: #123, #456
"""

from alembic import op
import sqlalchemy as sa

revision = 'YYYYMMDD_HHMMSS'
down_revision = 'previous_revision'
branch_labels = None
depends_on = None


def upgrade() -> None:
    """
    Apply migration with detailed step comments.
    """
    # Step 1: Create table
    # Purpose: Store patient consent records
    # Impact: Enables GDPR compliance tracking
    op.create_table(
        'patient_consents',
        sa.Column('id', sa.Integer, primary_key=True),
        sa.Column('patient_id', sa.Integer, sa.ForeignKey('patients.id')),
        sa.Column('consent_date', sa.DateTime(timezone=True)),
        # ... more columns
    )

    # Step 2: Create index
    # Purpose: Optimize consent date lookups (used in monthly reports)
    # Expected improvement: 50x faster queries
    op.execute("""
        CREATE INDEX CONCURRENTLY IF NOT EXISTS idx_patient_consent_date
        ON patient_consents (patient_id, consent_date);
    """)


def downgrade() -> None:
    """
    Rollback migration safely.

    WARNING: This will DELETE all consent records!

    op.drop_table('patient_consents')

## Common Patterns

### Safe Index Creation
```python
# ✅ Production-safe with CONCURRENTLY
op.execute("""
    CREATE INDEX CONCURRENTLY IF NOT EXISTS idx_name
    ON table_name (column_name);
""")

# ❌ DANGEROUS - locks table
op.create_index('idx_name', 'table_name', ['column_name'])
```

### Enum Modification
```python
# ✅ Safe enum creation
from sqlalchemy.dialects import postgresql

status_enum = postgresql.ENUM(
    'active', 'pending', 'completed',
    name='appointment_status',
    create_type=True
)
status_enum.create(op.get_bind(), checkfirst=True)
```

### Data Migration
```python
# ✅ Safe batch processing for large tables
from sqlalchemy import text

connection = op.get_bind()
batch_size = 1000
offset = 0

while True:
    result = connection.execute(text(f"""
        UPDATE patients
        SET normalized_email = LOWER(email)
        WHERE id IN (
            SELECT id FROM patients
            WHERE normalized_email IS NULL
            LIMIT {batch_size}
        )
    """))

    if result.rowcount == 0:
        break

    print(f"Processed batch at offset {offset}")
    offset += batch_size
```

## Risk Mitigation

### Risk: Delete migration that's in production
**Mitigation**: ALWAYS check `alembic_version` table first

### Risk: Break migration chain
**Mitigation**: Update `down_revision` in all dependent migrations

### Risk: Duplicate indexes in database
**Mitigation**: Check `pg_indexes` before creating new migrations

### Risk: Lose migration history
**Mitigation**: Never force-push to main, always use merge commits

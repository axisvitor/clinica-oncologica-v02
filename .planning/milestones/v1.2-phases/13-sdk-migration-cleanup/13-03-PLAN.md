    ```python
    def _safe_run_sync(self, prompt: str, deps: "AIDeps", *, operation: str) -> Any:
        """Celery-safe synchronous runner for pydantic-ai agents.

        pydantic-ai's run_sync() calls asyncio.get_event_loop() which may
        return a closed loop in Celery workers after cleanup cycles. This
        method ensures a fresh open loop before delegating.

        Use this from synchronous contexts (Celery tasks). For async contexts
        (FastAPI handlers), use _safe_run() instead.
        """
        import asyncio

        # Ensure event loop is open — Celery workers may have a closed loop
        # from worker_process_shutdown cleanup.
        try:
            loop = asyncio.get_event_loop()
            if loop.is_closed():
                loop = asyncio.new_event_loop()
                asyncio.set_event_loop(loop)
        except RuntimeError:
            # No current event loop in this thread
            loop = asyncio.new_event_loop()
            asyncio.set_event_loop(loop)

        from app.ai.pii_redaction import sanitize_prompt_text_for_external_ai
        from pydantic_ai.models.google import GoogleModel
        from pydantic_ai.providers.google import GoogleProvider

        safe_prompt = sanitize_prompt_text_for_external_ai(prompt)
        model = GoogleModel(
            deps.model_name,
            provider=GoogleProvider(api_key=deps.gemini_api_key),
        )
        result = self._agent.run_sync(safe_prompt, model=model, deps=deps)
        self._warn_on_output_pii(str(result.output), operation=operation)
        return result.output
    ```

    Also update `__init__.py` exports in `app/ai/agents/` if needed to expose the method cleanly.

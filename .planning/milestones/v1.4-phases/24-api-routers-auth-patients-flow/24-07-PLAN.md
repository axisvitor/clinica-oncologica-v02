Required async equivalents (from Phase 24 patterns in flows.py):
```python
# Read single: db.query(Model).filter(...).first()
result = await db.execute(select(Model).where(...))
obj = result.scalar_one_or_none()

# Read list: db.query(Model).filter(...).all()
result = await db.execute(select(Model).where(...))
objs = result.scalars().all()

# Read with join/eager: db.query(Model).join(Other).options(contains_eager(...))
stmt = select(Model).join(Other).options(contains_eager(...))
result = await db.execute(stmt)
# For unique results with joins:
objs = result.unique().scalars().all()

# Count: db.query(Model).filter(...).count()
result = await db.execute(select(func.count()).select_from(Model).where(...))
count_val = result.scalar_one()

# Scalar: db.query(Model.col).filter(...).scalar()
result = await db.execute(select(Model.col).where(...))
val = result.scalar_one_or_none()

# Bulk update: db.query(Model).filter(...).update({...})
await db.execute(update(Model).where(...).values(...))

# Write: db.add(obj) unchanged, but:
db.add(obj)
await db.flush()
await db.commit()
await db.refresh(obj)

# Delete:
await db.delete(obj)
await db.commit()

# Rollback:
await db.rollback()
```

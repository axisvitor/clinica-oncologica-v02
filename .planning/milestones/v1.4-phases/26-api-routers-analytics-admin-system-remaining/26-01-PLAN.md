```python
# DI swap
from app.core.database.async_engine import get_async_db
from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy import select, func, case

db: AsyncSession = Depends(get_async_db)

# Count query
result = await db.execute(select(func.count(Model.id)).where(...))
count = result.scalar() or 0

# Or using db.scalar shorthand:
count = await db.scalar(select(func.count(Model.id)).where(...)) or 0

# GROUP BY query
stmt = (
    select(Model.column, func.count(Model.id).label("count"))
    .where(...)
    .group_by(Model.column)
)
result = await db.execute(stmt)
rows = result.all()  # list of Row objects
for column_value, count in rows:
    ...

# JOIN + GROUP BY (analytics trend)
stmt = (
    select(
        func.extract("year", QuizSession.created_at).label("year"),
        func.extract("month", QuizSession.created_at).label("month"),
        func.count(QuizSession.id).label("total"),
        func.sum(case((QuizSession.status == "completed", 1), else_=0)).label("completed"),
    )
    .join(Patient, Patient.id == QuizSession.patient_id)
    .where(QuizSession.created_at >= start_date)
    .group_by(
        func.extract("year", QuizSession.created_at),
        func.extract("month", QuizSession.created_at),
    )
)
result = await db.execute(stmt)
rows = result.all()
for year, month, total, completed in rows:
    ...

# Outer join (patient engagement)
stmt = (
    select(Patient.id, func.count(QuizSession.id).label("quiz_count"))
    .outerjoin(QuizSession, Patient.id == QuizSession.patient_id)
    .where(...)
    .group_by(Patient.id)
)
result = await db.execute(stmt)
patient_quiz_counts = result.all()  # list of (id, quiz_count) rows
```

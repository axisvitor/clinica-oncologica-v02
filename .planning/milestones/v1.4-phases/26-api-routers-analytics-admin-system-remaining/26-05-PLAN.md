Standard async patterns:
```python
from app.core.database.async_engine import get_async_db
from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy import select, func, text

db: AsyncSession = Depends(get_async_db)

# Count query
result = await db.execute(select(func.count(Model.id)).where(...))
count = result.scalar() or 0

# Read single
result = await db.execute(select(Model).where(Model.id == id))
obj = result.scalar_one_or_none()

# Raw SQL health check (MUST await)
result = await db.execute(text("SELECT 1 as health_check"))
row = result.fetchone()
```

CRITICAL for database_health.py — Engine Pool Access Pattern:
```python
# Keep this import UNCHANGED:
from app.database import engine  # sync engine — for pool stats only

# The pool stats section is read-only — no session needed:

pool_size = pool.size()
checked_out = pool.checkedout()
overflow = pool.overflow()
```

The `check_database_health(db)` helper function must have ALL its `db.execute(text(...))` calls prefixed with `await`:
```python
# OLD: result = db.execute(text("SELECT 1 as health_check")).fetchone()
# NEW:
result = await db.execute(text("SELECT 1 as health_check"))
row = result.fetchone()

# OLD: version_result = db.execute(text("SELECT version_num FROM alembic_version")).fetchone()
# NEW:
version_result_obj = await db.execute(text("SELECT version_num FROM alembic_version"))
version_result = version_result_obj.fetchone()

# OLD: rls_result = db.execute(text("SELECT COUNT(*) ...")).fetchone()
# NEW:
rls_result_obj = await db.execute(text("SELECT COUNT(*) ..."))
rls_result = rls_result_obj.fetchone()
```
Since `check_database_health` is now using await, it must be declared as `async def`.

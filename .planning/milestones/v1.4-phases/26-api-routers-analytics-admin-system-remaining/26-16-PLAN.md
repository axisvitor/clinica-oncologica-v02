       class TestGlobalZeroSyncDI:

           def test_no_depends_get_db_in_any_router_file(self):
               """Scan ALL .py files under app/api/v2/routers/ for Depends(get_db)."""
               routers_dir = pathlib.Path("app/api/v2/routers")
               violations = []
               for py_file in routers_dir.rglob("*.py"):
                   if py_file.name == "__pycache__":
                       continue
                   content = py_file.read_text(encoding="utf-8", errors="ignore")
                   if "Depends(get_db)" in content:
                       violations.append(str(py_file))
               assert not violations, (
                   f"Depends(get_db) still present in {len(violations)} files: "
                   + ", ".join(violations)
               )

           def test_no_db_query_in_any_router_file(self):
               """Scan ALL .py files under app/api/v2/routers/ for db.query(."""
               routers_dir = pathlib.Path("app/api/v2/routers")
               violations = []
               # Exclude service files that may still use db.query for Celery/internal paths
               # Only check router endpoint files (not services/ subdirs unless in router package)
               for py_file in routers_dir.rglob("*.py"):

```python
def test_zero_db_query_in_registered_routers():
    """TEST-02: No db.query() in registered router modules (prevents MissingGreenlet)."""
    import importlib
    import inspect
    import pkgutil

    routers_pkg = importlib.import_module("app.api.v2.routers")
    router_dir = routers_pkg.__path__

    violations = []
    for importer, modname, ispkg in pkgutil.walk_packages(router_dir, prefix="app.api.v2.routers."):
        try:
            mod = importlib.import_module(modname)
            source = inspect.getsource(mod)
            # Check for db.query( which causes MissingGreenlet with AsyncSession
            if "db.query(" in source:
                violations.append(modname)
        except Exception:
            continue

    assert not violations, f"Router modules with db.query(): {violations}"
```

       ```python
           if is_opt_out_message(msg.text):
               await _process_opt_out(msg.phone, db)

           # Normal message -- downstream processing happens via Celery tasks or flow engine
       ```

       ```python
           try:

       async def _process_opt_out(phone: str, db: AsyncSession) -> None:
           """
           Process opt-out for patient identified by phone number.

           LGPD Art. 18: sets patient.messaging_stopped_at immediately.
           """
           try:
               from sqlalchemy import select
               from app.models.patient import Patient
               from app.utils.timezone import now_sao_paulo

               # Find patient by phone
               stmt = select(Patient).where(Patient.phone == phone)
               result = await db.execute(stmt)
               patient = result.scalar_one_or_none()

               if patient:
                   patient.messaging_stopped_at = now_sao_paulo()
                   await db.commit()
                   logger.info("WuzAPI opt-out processed for patient phone=%s", phone)
               else:
                   logger.warning("WuzAPI opt-out: no patient found for phone=%s", phone)
           except Exception as exc:
               logger.error("WuzAPI opt-out processing failed for phone=%s: %s", phone, exc, exc_info=True)
               try:
                   await db.rollback()
               except Exception:
                   pass
       ```

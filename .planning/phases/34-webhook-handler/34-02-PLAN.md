       ```python
       """
       WuzAPI webhook payload extractor.

       Maps WuzAPI/whatsmeow PascalCase JSON events to internal Python types.
       Handles both wrapped ({"event": {...}}) and flat payload formats defensively.
       """

       import re
       import logging
       from dataclasses import dataclass
       from typing import Any

       logger = logging.getLogger(__name__)


       @dataclass
       class WuzAPIInboundMessage:
           """Parsed inbound message from WuzAPI webhook."""
           message_id: str        # event.Info.ID
           phone: str             # Raw digits extracted from sender JID
           text: str              # Message text content
           is_lid: bool           # True if sender uses @lid addressing
           push_name: str = ""    # Sender's WhatsApp display name
           is_from_me: bool = False  # True if message sent by the connected account
           raw_message: dict | None = None  # Full Message dict for media detection


       @dataclass
       class WuzAPIReceiptEvent:
           """Parsed receipt event from WuzAPI webhook."""
           message_ids: list[str]  # MessageIDs from Receipt
           receipt_type: str       # whatsmeow ReceiptType string ("", "read", "played", etc.)
           sender_phone: str       # Raw digits from sender JID


       # whatsmeow ReceiptType -> internal status string
       # Source: go.mau.fi/whatsmeow/types ReceiptType constants (verified)
       RECEIPT_TYPE_TO_STATUS: dict[str, str] = {
           "":            "delivered",   # ReceiptTypeDelivered = "" (empty string)
           "sender":      "sent",        # ReceiptTypeSender
           "read":        "read",        # ReceiptTypeRead
           "read-self":   "read",        # ReceiptTypeReadSelf (treat as read)
           "played":      "played",      # ReceiptTypePlayed (audio/video)
           "played-self": "played",      # ReceiptTypePlayedSelf (treat as played)
           "retry":       "delivered",   # Retry delivery receipt
       }


       class WuzAPIMessageExtractor:
           """Extract structured data from WuzAPI webhook payloads."""

           @classmethod
           def extract_message(cls, payload: dict[str, Any]) -> WuzAPIInboundMessage | None:
               """
               Extract inbound message from WuzAPI Message event payload.

               Handles both wrapped and flat payload formats:
               - Wrapped: {"type": "Message", "event": {"Info": ..., "Message": ...}}
               - Flat: {"Info": ..., "Message": ...}

               Returns None if message ID or phone cannot be extracted.
               """
               event = payload.get("event") or payload
               info = event.get("Info") or {}
               message_id = info.get("ID") or ""
               if not message_id:
                   logger.warning("WuzAPI Message event missing Info.ID")
                   return None

               sender_jid = info.get("Sender") or ""
               is_lid = "@lid" in sender_jid
               phone = cls._jid_to_phone(sender_jid)
               if not phone:
                   logger.warning("WuzAPI Message event: could not extract phone from Sender=%r", sender_jid)
                   return None

               msg = event.get("Message") or {}
               text = (
                   msg.get("Conversation")
                   or ((msg.get("ExtendedTextMessage") or {}).get("Text"))
                   or ""
               )

               return WuzAPIInboundMessage(
                   message_id=message_id,
                   phone=phone,
                   text=text,
                   is_lid=is_lid,
                   push_name=info.get("PushName") or "",
                   is_from_me=bool(info.get("IsFromMe")),
                   raw_message=msg,
               )

           @classmethod
           def extract_receipt(cls, payload: dict[str, Any]) -> WuzAPIReceiptEvent | None:
               """
               Extract receipt event from WuzAPI ReadReceipt payload.

               Returns None if no message IDs can be extracted.
               """
               event = payload.get("event") or payload
               info = event.get("Info") or {}
               receipt = event.get("Receipt") or {}

               sender_jid = info.get("Sender") or ""
               phone = cls._jid_to_phone(sender_jid)

               # CRITICAL: ReceiptType "" (empty string) means Delivered
               # Do NOT use `if not receipt_type:` -- that would filter out Delivered
               receipt_type = receipt.get("Type")
               if receipt_type is None:
                   receipt_type = ""

               # MessageIDs list; fallback to single ID from Info
               message_ids = receipt.get("MessageIDs") or []
               if not message_ids:
                   fallback_id = info.get("ID")
                   if fallback_id:
                       message_ids = [fallback_id]
               message_ids = [m for m in message_ids if m]

               if not message_ids:
                   logger.warning("WuzAPI Receipt event: no message IDs found")
                   return None

               return WuzAPIReceiptEvent(
                   message_ids=message_ids,
                   receipt_type=receipt_type,
                   sender_phone=phone,
               )

           @classmethod
           def _jid_to_phone(cls, jid: str) -> str:
               """
               Extract raw phone digits from a WhatsApp JID.

               Handles:
               - Standard: "5511987654321@s.whatsapp.net" -> "5511987654321"
               - LID: "12345@lid" -> "12345"
               - AD JID: "5511987654321.0:1@s.whatsapp.net" -> "5511987654321"
               - Plain digits: "5511987654321" -> "5511987654321"
               """
               if not jid:
                   return ""
               user = jid.split("@")[0] if "@" in jid else jid
               # AD JID format: user.agent:device -- extract user part before "."
               if ":" in user and "." in user:
                   user = user.split(".")[0]
               elif ":" in user:
                   user = user.split(":")[0]
               return re.sub(r"[^\d]", "", user)
       ```

<objective>
Validate template day_config at flow start so malformed configs fail fast with clear errors.

Purpose: Currently, `load_flow_context` in `_flow_message_flow.py` does minimal validation of the day_config returned by `handler._get_day_config()`. It checks if it is a dict, if messages is a list, and if send_mode is valid. But it does not validate the structure of individual messages (e.g., missing `content` field, invalid `expects_response` type) or detect completely empty configs that should not silently skip. This plan creates a dedicated `validate_day_config` function that performs thorough validation and is called early in `load_flow_context`, failing fast with a `DayConfigValidationError` that includes specific details about what is wrong.

Output: New `config_validation.py` module, modified `_flow_message_flow.py` with early validation, and comprehensive tests.
</objective>

```python
"""Validation for flow template day_config at flow start."""

from __future__ import annotations

import logging
from typing import Any, Dict, List

from app.services.flow._flow_orchestration_utils import _CANONICAL_SEND_MODES

logger = logging.getLogger(__name__)


class DayConfigValidationError(ValueError):
    """Raised when day_config fails structural validation."""

    def __init__(self, message: str, errors: list[str] | None = None):
        self.errors = errors or [message]
        super().__init__(message)


def validate_day_config(
    day_config: Any,
    *,
    flow_kind: str = "",
    day_number: int = 0,
) -> Dict[str, Any]:
    """
    Validate day_config structure at flow start.

    Raises DayConfigValidationError with specific details if validation fails.
    Returns the validated day_config dict on success.
    """
    errors: list[str] = []

    if day_config is None:
        raise DayConfigValidationError(
            f"day_config is None for flow_kind={flow_kind} day={day_number}",
            errors=["day_config is None"],
        )

    if not isinstance(day_config, dict):
        raise DayConfigValidationError(
            f"day_config must be a dict, got {type(day_config).__name__}",
            errors=[f"day_config type: {type(day_config).__name__}"],
        )

    # Validate send_mode
    send_mode = day_config.get("send_mode", "single")
    if not isinstance(send_mode, str):
        errors.append(f"send_mode must be a string, got {type(send_mode).__name__}")
    elif send_mode.strip().lower() not in _CANONICAL_SEND_MODES:
        allowed = ", ".join(sorted(_CANONICAL_SEND_MODES))
        errors.append(f"Invalid send_mode '{send_mode}'. Allowed: {allowed}")

    # Validate messages
    messages = day_config.get("messages")
    if messages is None:
        errors.append("'messages' key is missing from day_config")
    elif not isinstance(messages, list):
        errors.append(f"'messages' must be a list, got {type(messages).__name__}")
    elif len(messages) == 0:
        errors.append("'messages' list is empty — no messages to send")
    else:
        for i, msg in enumerate(messages):
            if not isinstance(msg, dict):
                errors.append(f"messages[{i}] must be a dict, got {type(msg).__name__}")
                continue
            content = msg.get("content")
            if content is None:
                errors.append(f"messages[{i}] missing required 'content' key")
            elif not isinstance(content, str):
                errors.append(f"messages[{i}].content must be a string, got {type(content).__name__}")
            elif not content.strip():
                errors.append(f"messages[{i}].content is empty")

    if errors:
        raise DayConfigValidationError(
            f"day_config validation failed for flow_kind={flow_kind} day={day_number}: {'; '.join(errors)}",
            errors=errors,
        )

    return day_config
```

1. Add a new job `smoke-adk` between the existing `test-backend` and `build-backend` jobs:

```yaml
smoke-adk:
  name: ADK Smoke Tests
  runs-on: ubuntu-latest
  timeout-minutes: 5
  needs: [lint-backend]

  steps:
    - name: Checkout code
      uses: actions/checkout@v4

    - name: Set up Python
      uses: actions/setup-python@v5
      with:
        python-version: ${{ env.PYTHON_VERSION }}
        cache: 'pip'

    - name: Cache Python dependencies
      uses: actions/cache@v4
      with:
        path: |
          ~/.cache/pip
          backend-hormonia/.venv
        key: ${{ runner.os }}-pip-smoke-${{ hashFiles('backend-hormonia/requirements*.txt') }}
        restore-keys: |
          ${{ runner.os }}-pip-smoke-
          ${{ runner.os }}-pip-

    - name: Install dependencies
      working-directory: backend-hormonia
      run: |
        python -m pip install --upgrade pip
        pip install pytest pytest-asyncio pytest-mock
        if [ -f requirements.txt ]; then pip install -r requirements.txt; fi
        if [ -f requirements-dev.txt ]; then pip install -r requirements-dev.txt; fi

    - name: Run ADK smoke tests
      working-directory: backend-hormonia
      env:
        TESTING: "true"
        ENVIRONMENT: test
        WHATSAPP_WUZAPI_TOKEN: smoke-ci-token
      run: |
        pytest -m adk_smoke --tb=short -v --junitxml=adk-smoke-report.xml

    - name: Upload smoke test results
      uses: actions/upload-artifact@v4
      if: always()
      with:
        name: adk-smoke-results
        path: backend-hormonia/adk-smoke-report.xml
        retention-days: 7
```

Key design decisions:
- `needs: [lint-backend]` so it runs in parallel with `test-backend` (not after it) for faster feedback.
- No `services` block (no postgres, no redis) because smoke tests use in-memory fallbacks.
- Sets `WHATSAPP_WUZAPI_TOKEN=smoke-ci-token` to prevent settings bootstrap validation failure.
- Sets `TESTING=true` and `ENVIRONMENT=test` to match existing test-backend env pattern.
- `timeout-minutes: 5` because smoke tests should complete in under 30 seconds; 5 minutes is generous safety margin.
- Uploads JUnit XML report as artifact for per-scenario visibility.

3. Update the `ci-status` job's `needs` to include `smoke-adk`:
